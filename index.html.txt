    <script>
        function timeSince(date) {
            const now = new Date();
            const seconds = Math.floor((now - date) / 1000);
            const minutes = Math.floor(seconds / 60);
            const hours = Math.floor(minutes / 60);
            const days = Math.floor(hours / 24);
            const years = Math.floor(days / 365);

            return {
                years: years,
                days: days % 365,
                hours: hours % 24,
                minutes: minutes % 60,
                seconds: seconds % 60
            };
        }

        function updateCounter() {
            const birthDate = new Date('1987-08-16'); // Cambia esta fecha si es necesario
            const time = timeSince(birthDate);

            const counterText = ${time.years} Años, ${time.days} Días, ${time.hours} Horas, ${time.minutes} Minutos, ${time.seconds} Segundos;
            document.getElementById("counter").textContent = counterText;
        }

        setInterval(updateCounter, 1000);
        updateCounter();
    </script>
